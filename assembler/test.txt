.data
_s1: .asciiz "Hello, World!"

.text 
.globl main 

main:
addiu $sp, $sp, -16
jal label2

li $v0, 10
syscall

label1:
sw $ra, 0($sp)

li $2, 4
lw $4, 4($sp)
syscall

li $4, 10
li $2, 11
syscall

lw $ra, 0($sp)
addiu $sp, $sp, 12
jr $ra

label2:
sw $ra, 0($sp)

la $8, _s1
sw $8, 8($sp)

li $9, 0
li $10, -8
add $10, $10, $sp
li $11, 8
add $11, $11, $sp
add $10, $10, $9
add $11, $11, $9
lw $8, ($11)
sw $8, ($10)
addi $9, $9, 4

addiu $sp, $sp, -12
jal label1

lw $8 -4($sp)
sw $8 4($sp)

lw $ra, 0($sp)
addiu $sp, $sp, 16
jr $ra
